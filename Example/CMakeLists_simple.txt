cmake_minimum_required(VERSION 3.16)
project(SimpleExample VERSION 1.0.0 LANGUAGES CXX)

# Configure the C++ standard requirements
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Force all toolchains to use C++20
if(MSVC)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /std:c++20")
else()
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++20")
endif()

# Include project headers
include_directories(${CMAKE_SOURCE_DIR}/Inc)

# Build the sample executable
add_executable(simple_example simple_example.cpp)

# Send artifacts to the project Bin directory
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/Bin/x86_64/${CMAKE_BUILD_TYPE})
